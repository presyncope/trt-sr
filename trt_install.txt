# 1. 파일 다운로드
> wget https://developer.download.nvidia.com/compute/tensorrt/10.14.1/local_installers/nv-tensorrt-local-repo-ubuntu2404-10.14.1-cuda-13.0_1.0-1_amd64.deb

# 2. 로컬 저장소 패키지 설치
> sudo dpkg -i nv-tensorrt-local-repo-ubuntu2404-10.14.1-cuda-13.0_1.0-1_amd64.deb

# 3. 로컬 저장소의 GPG 키 복사 (중요: 설치를 위해 시스템에 키를 등록해야 함)
# 설치 후 터미널에 출력되는 안내 메시지의 경로를 확인하세요. 보통 아래와 같습니다.
> sudo cp /var/nv-tensorrt-local-repo-ubuntu2404-10.14.1-cuda-13.0/nv-tensorrt-local-D12DA6BC-keyring.gpg /usr/share/keyrings/

# troubleshooting (정확한 gpg 파일 경로 찾기)
> ls /var/ | grep nv-tensorrt-local-repo 
nv-tensorrt-local-repo-ubuntu2404-10.14.1-cuda-13.0

> ls /var/nv-tensorrt-local-repo-ubuntu2404-10.14.1-cuda-13.0/
... nv-tensorrt-local-D12DA6BC-keyring.gpg

# 4. 패키지 목록 업데이트 및 설치
> sudo apt-get update
> sudo apt-get install -y tensorrt

# 5. Python 및 ONNX 파서 설치를
> sudo apt-get install -y python3-libnvinfer-dev
