# TRT 10.15.1 - ubuntu 24.04 - CUDA 13.0~1 설치 방법
```bash
wget https://developer.download.nvidia.com/compute/tensorrt/10.15.1/local_installers/nv-tensorrt-local-repo-ubuntu2404-10.15.1-cuda-13.1_1.0-1_amd64.deb
sudo dpkg -i nv-tensorrt-local-repo-ubuntu2404-10.15.1-cuda-13.1_1.0-1_amd64.deb
sudo cp /var/nv-tensorrt-local-repo-ubuntu2404-10.15.1-cuda-13.1/*.gpg /usr/share/keyrings/
sudo apt-get update
sudo apt-get install tensorrt
```
